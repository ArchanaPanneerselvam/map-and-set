                    let a="abcadeecfb"
                    let ans=new Map()
                    for(let str of a){
                        if(ans.has(str)){
                            ans.set(str,ans.get(str)+1)
                        }
                        else{
                            ans.set(str,1)
                        }
                    }
                    for(let [key,value] of ans){
